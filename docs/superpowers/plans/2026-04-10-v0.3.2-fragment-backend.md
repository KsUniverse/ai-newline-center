- [ ] **Step 3: Create `FragmentRepository` with cursor pagination and precise batch create**

Create `src/server/repositories/fragment.repository.ts`:

```ts
import type { Prisma, PrismaClient } from "@prisma/client";

import { prisma } from "@/lib/prisma";
import type { CursorPaginatedData } from "@/types/api";

type DatabaseClient = PrismaClient | Prisma.TransactionClient;

const fragmentInclude = {
  createdByUser: {
    select: {
      id: true,
      name: true,
    },
  },
} satisfies Prisma.FragmentInclude;

function encodeCursor(createdAt: Date, id: string): string {
  return Buffer.from(
    JSON.stringify({ c: createdAt.toISOString(), i: id }),
    "utf8",
  ).toString("base64url");
}

function decodeCursor(cursor: string): { c: Date; i: string } {
  const parsed = JSON.parse(Buffer.from(cursor, "base64url").toString("utf8")) as {
    c: string;
    i: string;
  };

  return {
    c: new Date(parsed.c),
    i: parsed.i,
  };
}

class FragmentRepository {
  async findMany(params: {
    organizationId: string;
    q?: string;
    cursor?: string;
    limit: number;
  }): Promise<CursorPaginatedData<Prisma.FragmentGetPayload<{ include: typeof fragmentInclude }>>> {
    const { organizationId, q, cursor, limit } = params;
    const decoded = cursor ? decodeCursor(cursor) : null;

    const items = await prisma.fragment.findMany({
      where: {
        organizationId,
        deletedAt: null,
        ...(q ? { content: { contains: q } } : {}),
        ...(decoded
          ? {
              OR: [
                { createdAt: { lt: decoded.c } },
                { AND: [{ createdAt: decoded.c }, { id: { lt: decoded.i } }] },
              ],
            }
          : {}),
      },
      include: fragmentInclude,
      orderBy: [{ createdAt: "desc" }, { id: "desc" }],
      take: limit + 1,
    });

    const hasMore = items.length > limit;
    const pageItems = hasMore ? items.slice(0, limit) : items;
    const lastItem = pageItems.at(-1);

    return {
      items: pageItems,
      hasMore,
      nextCursor: hasMore && lastItem ? encodeCursor(lastItem.createdAt, lastItem.id) : null,
    };
  }

  async createManyPrecisely(
    items: Array<{ content: string; organizationId: string; createdByUserId: string }>,
  ) {
    return prisma.$transaction(async (tx) => {
      const created: Prisma.FragmentGetPayload<{ include: typeof fragmentInclude }>[] = [];
      for (const item of items) {
        created.push(
          await tx.fragment.create({
            data: item,
            include: fragmentInclude,
          }),
        );
      }
      return created;
    });
  }

  async findById(id: string) {
    return prisma.fragment.findFirst({
      where: {
        id,
        deletedAt: null,
      },
      include: fragmentInclude,
    });
  }

  async softDelete(data: { id: string; organizationId: string }) {
    return prisma.fragment.updateMany({
      where: {
        id: data.id,
        organizationId: data.organizationId,
        deletedAt: null,
      },
      data: {
        deletedAt: new Date(),
      },
    });
  }
}

export const fragmentRepository = new FragmentRepository();
```

- [ ] **Step 4: Create `FragmentService` with validation, DTO mapping, and delete authorization**

Create `src/server/services/fragment.service.ts`:

```ts
import { UserRole } from "@prisma/client";

import { AppError } from "@/lib/errors";
import { fragmentRepository } from "@/server/repositories/fragment.repository";
import type { CreateFragmentsInput, CreateFragmentsResult, FragmentDTO, ListFragmentsParams } from "@/types/fragment";
import type { CursorPaginatedData } from "@/types/api";
import type { SessionUser } from "@/types/session";

class FragmentService {
  private toDto(fragment: {
    id: string;
    content: string;
    organizationId: string;
    createdByUserId: string;
    createdAt: Date;
    createdByUser: { id: string; name: string | null };
  }): FragmentDTO {
    return {
      id: fragment.id,
      content: fragment.content,
      organizationId: fragment.organizationId,
      createdByUserId: fragment.createdByUserId,
      createdByUser: {
        id: fragment.createdByUser.id,
        name: fragment.createdByUser.name ?? "未知用户",
      },
      createdAt: fragment.createdAt.toISOString(),
    };
  }

  async listFragments(
    caller: SessionUser,
    params: ListFragmentsParams,
  ): Promise<CursorPaginatedData<FragmentDTO>> {
    const result = await fragmentRepository.findMany({
      organizationId: caller.organizationId,
      q: params.q,
      cursor: params.cursor,
      limit: params.limit ?? 20,
    });

    return {
      ...result,
      items: result.items.map((item) => this.toDto(item)),
    };
  }

  async createFragments(
    caller: SessionUser,
    input: CreateFragmentsInput,
  ): Promise<CreateFragmentsResult> {
    const normalized = input.contents
      .map((content) => content.trim())
      .filter((content) => content.length > 0);

    if (normalized.length === 0) {
      throw new AppError("VALIDATION_ERROR", "请至少输入一条有效观点", 400);
    }

    if (normalized.some((content) => content.length > 500)) {
      throw new AppError("VALIDATION_ERROR", "每条观点不能超过 500 字", 400);
    }

    const items = await fragmentRepository.createManyPrecisely(
      normalized.map((content) => ({
        content,
        organizationId: caller.organizationId,
        createdByUserId: caller.id,
      })),
    );

    return {
      created: items.length,
      items: items.map((item) => this.toDto(item)),
    };
  }

  async deleteFragment(caller: SessionUser, fragmentId: string): Promise<{ id: string }> {
    const fragment = await fragmentRepository.findById(fragmentId);
    if (!fragment) {
      throw new AppError("NOT_FOUND", "观点不存在", 404);
    }

    if (fragment.organizationId !== caller.organizationId) {
      throw new AppError("FORBIDDEN", "无操作权限", 403);
    }

    if (
      caller.role === UserRole.EMPLOYEE &&
      fragment.createdByUserId !== caller.id
    ) {
      throw new AppError("FORBIDDEN", "无操作权限", 403);
    }

    await fragmentRepository.softDelete({
      id: fragmentId,
      organizationId: caller.organizationId,
    });

    return { id: fragmentId };
  }
}

export const fragmentService = new FragmentService();
```

```ts
import { UserRole } from "@prisma/client";
import { z } from "zod";

import { auth } from "@/lib/auth";
import { requireRole } from "@/lib/auth-guard";
import { handleApiError, successResponse } from "@/lib/api-response";
import { fragmentService } from "@/server/services/fragment.service";

const listQuerySchema = z.object({
  q: z.string().max(200).optional(),
  cursor: z.string().optional(),
  limit: z.coerce.number().int().min(1).max(100).default(20),
});

const createSchema = z.object({
  contents: z.array(z.string().min(1).max(500)).min(1).max(50),
});

export async function GET(request: Request) {
  try {
    const session = await auth();
    requireRole(session, UserRole.EMPLOYEE, UserRole.BRANCH_MANAGER, UserRole.SUPER_ADMIN);

    const query = listQuerySchema.parse(Object.fromEntries(new URL(request.url).searchParams));
    const result = await fragmentService.listFragments(session.user, query);

    return successResponse(result);
  } catch (error) {
    return handleApiError(error);
  }
}

export async function POST(request: Request) {
  try {
    const session = await auth();
    requireRole(session, UserRole.EMPLOYEE, UserRole.BRANCH_MANAGER, UserRole.SUPER_ADMIN);

    const body = await request.json();
    const data = createSchema.parse(body);
    const result = await fragmentService.createFragments(session.user, data);

    return successResponse(result, 201);
  } catch (error) {
    return handleApiError(error);
  }
}
```
